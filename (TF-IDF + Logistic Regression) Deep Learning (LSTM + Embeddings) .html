Preprocess Sentiment Labels¶

sentiment labels in the "Reviews" column are not numeric, we need to categorize them into Positive, Negative, or Neutral.

In [6]:
import matplotlib.pyplot as plt
import seaborn as sns
import os

# Define a function to classify sentiment
positive_words = ['good', 'great', 'excellent', 'amazing', 'wonderful', 'positive', 'best']
negative_words = ['bad', 'terrible', 'awful', 'poor', 'negative', 'worst']

def assign_sentiment(review):
    review = str(review).lower()
    if any(word in review for word in positive_words):
        return "Positive"
    elif any(word in review for word in negative_words):
        return "Negative"
    else:
        return "Neutral"  # Default if no keywords found

# Apply sentiment function
df['Sentiment'] = df['Reviews'].apply(assign_sentiment)

# Show sentiment distribution
print("Sentiment Distribution:\n", df['Sentiment'].value_counts())

# Plot distribution
plt.figure(figsize=(8, 6))
sns.countplot(data=df, x='Sentiment', palette='flare').set_title('Sentiment Distribution')

Text Preprocessing¶

Removing punctuation & converting to lowercase. Removing stopwords . Lemmatizing words converting words to their base

In [4]:
def cleaning(text):
    # Remove punctuation and convert to lowercase
    text = text.translate(str.maketrans('', '', string.punctuation)).lower()
    
    # Remove stopwords
    words = [word for word in text.split() if word not in stopwords.words('english')]
    
    # Lemmatize words
    lemmatizer = WordNetLemmatizer()
    words = [lemmatizer.lemmatize(word, 'v') for word in words]
    
    return ' '.join(words)

# Apply cleaning function to the dataset
df['Cleaned_Reviews'] = df['Reviews'].apply(cleaning)

# Tokenization for LSTM
tokenizer = Tokenizer(num_words=50000, oov_token='<OOV>')
tokenizer.fit_on_texts(df['Cleaned_Reviews'])  # Fit tokenizer on raw text

# Use y_train.index to extract corresponding text data from df
train_texts = df.loc[y_train.index, 'Cleaned_Reviews']
test_texts = df.loc[y_test.index, 'Cleaned_Reviews']

# Convert raw text data to sequences
train_seq = tokenizer.texts_to_sequences(train_texts)
test_seq = tokenizer.texts_to_sequences(test_texts)

# Pad sequences to a fixed length
train_padded = pad_sequences(train_seq, maxlen=100, padding='post')
test_padded = pad_sequences(test_seq, maxlen=100, padding='post')

# One-hot encode sentiment labels
lb = LabelBinarizer()
train_labels = lb.fit_transform(y_train)
test_labels = lb.transform(y_test)

# Define the LSTM model
model = tf.keras.models.Sequential([
    tf.keras.layers.Embedding(input_dim=len(tokenizer.word_index) + 1, output_dim=8),
    tf.keras.layers.Bidirectional(tf.keras.layers.LSTM(16)),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(8, activation='relu', kernel_regularizer=l2(0.001), bias_regularizer=l2(0.001)),
    tf.keras.layers.Dropout(0.5),
    tf.keras.layers.Dense(3, activation='softmax')  # 3 classes (Positive, Neutral, Negative)
])

# Compile the model
model.compile(optimizer=tf.optimizers.Adam(learning_rate=0.0001), loss='categorical_crossentropy', metrics=['accuracy'])

# Train the model
model.fit(train_padded, train_labels, epochs=25, validation_data=(test_padded, test_labels))

# Save the trained model
model.save('dl_model.h5')
